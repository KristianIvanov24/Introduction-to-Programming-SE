### Сложност на алгоритми

``` c++
int getFirst(int arr[]) {
    return arr[0];
}
```

```c++
int sum(const int*, int n) {
    int s = 0;
    for (int i = 0; i < n; i++)
        s += arr[i];
    return s;
}
```

```c++
void print(const int* arr, int n) {
    for (int i = 0; i < n; i++)
        cout << arr[i];

    for (int i = 0; i < n; i++)
        cout << arr[i];
}
```

```c++
void pairs(int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cout << i << j;
}
```

```c++
void triangle(int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j <= i; j++)
            cout << "*";
}
```

```c++
int binarySearch(const int* arr, int n, int x) {
    int l = 0, r = n - 1;

    while (l <= r) {
        int mid = (l + r) / 2;
        if (arr[mid] == x) return mid;
        else if (x < arr[mid]) r = mid - 1;
        else l = mid + 1;
    }
    return -1;
}
```

```c++

    for (int i = 0; i < n; i += 2)
        cout << i;
}
```
